%matplotlib inline
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

from IPython.html.widgets import interact

@interact
def explore(n=10):
    x = np.linspace(0, n, 1000)
    plt.plot(x, np.sin(x), x, np.cos(x));
    sns.set()
    plt.plot(x, np.sin(x), x, np.cos(x));

About
